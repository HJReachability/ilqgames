37 ///////////////////////////////////////////////////////////////////////////////
38 //
39 // Core open-loop LQ game solver based on Basar and Olsder, Chapter 6. All
40 // notation matches the text, though we shall assume that `c` (additive drift in
41 // dynamics) is always `0`, which holds because these dynamics are for delta x,
42 // delta us. Also, we have modified terms slightly to account for linear terms
43 // in the stage cost for control, i.e.
44 // control penalty i = 0.5 \sum_j du_j^T R_ij (du_j + 2 r_ij)
45 //
46 // Solve a time-varying, finite horizon LQ game (finds open-loop Nash
47 // feedback strategies for both players).
48 //
49 // Assumes that dynamics are given by
50 // ``` dx_{k+1} = A_k dx_k + \sum_i Bs[i]_k du[i]_k ```
51 //
52 // Returns strategies Ps, alphas. Here, all the Ps are zero (by default), and
53 // only the alphas are nonzero.
54 //
55 // Notation is based on derivation which may be found in the PDF included in
56 // this repository named "open_loop_lq_derivation.pdf".
57 //

73 std::vector<Strategy> LQOpenLoopSolver::Solve(
74  const std::vector<LinearDynamicsApproximation>& linearization,
75  const std::vector<std::vector<QuadraticCostApproximation>>&
76  quadraticization,
77  const VectorXf& x0) {
78  CHECK_EQ(linearization.size(), num_time_steps_);
79  CHECK_EQ(quadraticization.size(), num_time_steps_);
80 
81  // List of player-indexed strategies (each of which is a time-indexed
82  // affine state error-feedback controller). Since this is an open-loop
83  // strategy, we will not change the default zero value of the P matrix.
84  std::vector<Strategy> strategies;
85  for (PlayerIndex ii = 0; ii < dynamics_->NumPlayers(); ii++)
86  strategies.emplace_back(num_time_steps_, dynamics_->XDim(),
87  dynamics_->UDim(ii));
88 
89  // Initialize m^i and M^i and index first by time and then by player.
90  for (PlayerIndex ii = 0; ii < dynamics_->NumPlayers(); ii++) {
91  ms_.back()[ii] = quadraticization.back()[ii].state.grad;
92  Ms_.back()[ii] = quadraticization.back()[ii].state.hess;
93  }
94 
95  // (1) Work backward in time and cache "special" terms.
96  // NOTE: time starts from the second-to-last entry since we'll treat the
97  // final entry as a terminal cost as in Basar and Olsder, ch. 6.
98  for (int kk = num_time_steps_ - 2; kk >= 0; kk--) {
99  // Unpack linearization and quadraticization at this time step.
100  const auto& lin = linearization[kk];
101  const auto& quad = quadraticization[kk];
102  const auto& next_quad = quadraticization[kk + 1];
103 
104  // Campute capital lambdas.
105  capital_lambdas_[kk] =
106  MatrixXf::Identity(dynamics_->XDim(), dynamics_->XDim());
107  for (PlayerIndex ii = 0; ii < dynamics_->NumPlayers(); ii++) {
108  const auto control_iter = quad[ii].control.find(ii);
109  CHECK(control_iter != quad[ii].control.end());
110 
111  chol_Rs_[kk][ii].compute(control_iter->second.hess);
112  warped_Bs_[kk][ii] = chol_Rs_[kk][ii].solve(lin.Bs[ii].transpose());
113  warped_rs_[kk][ii] = chol_Rs_[kk][ii].solve(control_iter->second.grad);
114  capital_lambdas_[kk] += lin.Bs[ii] * warped_Bs_[kk][ii] * Ms_[kk + 1][ii];
115  }
116 
117  // Compute inv(capital lambda).
118  qr_capital_lambdas_[kk].compute(capital_lambdas_[kk]);
119 
120  // Compute Ms and ms.
121  for (PlayerIndex ii = 0; ii < dynamics_->NumPlayers(); ii++) {
122  Ms_[kk][ii] =
123  quad[ii].state.hess + lin.A.transpose() * Ms_[kk + 1][ii] *
124  qr_capital_lambdas_[kk].solve(lin.A);
125 
126  // Intermediate term in ms computation.
127  VectorXf intermediary = VectorXf::Zero(dynamics_->XDim());
128  for (PlayerIndex jj = 0; jj < dynamics_->NumPlayers(); jj++) {
129  intermediary -= lin.Bs[jj] * (warped_Bs_[kk][jj] * ms_[kk + 1][ii] +
130  warped_rs_[kk][jj]);
131  }
132 
133  ms_[kk][ii] =
134  next_quad[ii].state.grad +
135  lin.A.transpose() *
136  (ms_[kk + 1][ii] +
137  Ms_[kk + 1][ii] * qr_capital_lambdas_[kk].solve(intermediary));
138  }
139  }
140 
141  // (2) Now compute optimal state and control trajectory forward in time.
142  VectorXf x_star = x0;
143  VectorXf last_x_star = x_star;
144  for (size_t kk = 0; kk < num_time_steps_ - 1; kk++) {
145  // Unpack linearization at this time step.
146  const auto& lin = linearization[kk];
147 
148  // Intermediate term in u and x computations.
149  VectorXf intermediary = lin.A * x_star;
150  for (PlayerIndex ii = 0; ii < dynamics_->NumPlayers(); ii++) {
151  intermediary -= lin.Bs[ii] * (warped_Bs_[kk][ii] * ms_[kk + 1][ii] +
152  warped_rs_[kk][ii]);
153  }
154 
155  // Compute optimal x.
156  last_x_star = x_star;
157  x_star = qr_capital_lambdas_[kk].solve(intermediary);
158 
159  // Compute optimal u and store (sign flipped) in alpha.
160  for (PlayerIndex ii = 0; ii < dynamics_->NumPlayers(); ii++) {
161  strategies[ii].alphas[kk] =
162  warped_Bs_[kk][ii] * (Ms_[kk + 1][ii] * x_star + ms_[kk + 1][ii]);
163  }
164 
165  // Check dynamic feasibility.
166  // VectorXf check_x = lin.A * last_x_star;
167  // for (PlayerIndex ii = 0; ii < dynamics_->NumPlayers(); ii++)
168  // check_x -= lin.Bs[ii] * strategies[ii].alphas[kk];
169 
170  // CHECK_LE((x_star - check_x).cwiseAbs().maxCoeff(), 1e-1);
171  }
172 
173  return strategies;
