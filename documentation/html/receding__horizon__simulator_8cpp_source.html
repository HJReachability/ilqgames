39 // Utility for solving a problem using a receding horizon, simulating dynamics
40 // forward at each stage to account for the passage of time.
41 //
42 // This class is intended as a facsimile of a real-time, online receding horizon
43 // problem in which short horizon problems are solved asynchronously throughout
44 // operation.
45 //
46 ///////////////////////////////////////////////////////////////////////////////

63 using clock = std::chrono::system_clock;
64 
65 std::vector<std::shared_ptr<const SolverLog>> RecedingHorizonSimulator(
66  Time final_time, Time planner_runtime, Problem* problem) {
67  CHECK_NOTNULL(problem);
68 
69  // Set up a list of solver logs, one per solver invocation.
70  std::vector<std::shared_ptr<const SolverLog>> logs;
71 
72  // Initial run of the solver. Keep track of time in order to know how much to
73  // integrate dynamics forward.
74  auto solver_call_time = clock::now();
75  logs.push_back(problem->Solve());
76  Time elapsed_time =
77  std::chrono::duration<Time>(clock::now() - solver_call_time).count();
78 
79  VLOG(0) << "Solved initial problem in " << elapsed_time << " seconds, with "
80  << logs.back()->NumIterates() << " iterations.";
81  const auto& dynamics = problem->Solver().Dynamics();
82 
83  // Keep a solution splicer to incorporate new receding horizon solutions.
84  SolutionSplicer splicer(*logs.front());
85 
86  // Repeatedly integrate dynamics forward, reset problem initial conditions,
87  // and resolve.
88  VectorXf x(problem->InitialState());
89  Time t = splicer.CurrentOperatingPoint().t0;
90 
91  while (true) {
92  // Break the loop if it's been long enough.
93  // Integrate a little more.
94  constexpr Time kExtraTime = 0.05;
95  t += kExtraTime; // + planner_runtime;
96 
97  if (t >= final_time || !splicer.ContainsTime(t + planner_runtime +
98  problem->Solver().TimeStep()))
99  break;
100 
101  x = dynamics.Integrate(t - kExtraTime, t, x,
102  splicer.CurrentOperatingPoint(),
103  splicer.CurrentStrategies());
104 
105  // Overwrite problem with spliced solution.
106  problem->OverwriteSolution(splicer.CurrentOperatingPoint(),
107  splicer.CurrentStrategies());
108 
109  // Set up next receding horizon problem and solve.
110  problem->SetUpNextRecedingHorizon(x, t, planner_runtime);
111 
112  solver_call_time = clock::now();
113  logs.push_back(problem->Solve(planner_runtime));
114  elapsed_time =
115  std::chrono::duration<Time>(clock::now() - solver_call_time).count();
116 
117  CHECK_LE(elapsed_time, planner_runtime);
118  VLOG(0) << "t = " << t << ": Solved warm-started problem in "
119  << elapsed_time << " seconds.";
120 
121  // Break the loop if it's been long enough.
122  t += elapsed_time;
123  if (t >= final_time || !splicer.ContainsTime(t)) break;
124 
125  // Integrate dynamics forward to account for solve time.
126  x = dynamics.Integrate(t - elapsed_time, t, x,
127  splicer.CurrentOperatingPoint(),
128  splicer.CurrentStrategies());
129 
130  // Add new solution to splicer if it converged.
131  if (logs.back()->WasConverged()) splicer.Splice(*logs.back());
132  }
133 
134  return logs;
135 }
