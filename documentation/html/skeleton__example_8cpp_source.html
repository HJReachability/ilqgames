48 //
49 // Steps (some of these are already done in this example but can be modified as
50 // suggested below):
51 // 1. Set the cost weights.
52 // 2. Set the nominal speed for each car.

132 SkeletonExample::SkeletonExample(const SolverParams& params) {
133  // Create dynamics. In this case, we have two cars with decoupled dynamics
134  // (they are only coupled through the cost structure of the game). This is
135  // expressed in the ConcatenatedDynamicalSystem class. Here, each player's
136  // dynamics follow that of a standard 5D bicycle model with inter-axle
137  // distance below.
138  static constexpr float kInterAxleDistance = 4.0; // m
139  const std::shared_ptr<const ConcatenatedDynamicalSystem> dynamics(
140  new ConcatenatedDynamicalSystem(
141  {std::make_shared<P1>(kInterAxleDistance),
142  std::make_shared<P2>(kInterAxleDistance)},
143  kTimeStep));
144 
145  // Set up initial state. Initially, this is zero, but then we override
146  // individual dimensions to match the desired initial conditions above.
147  x0_ = VectorXf::Zero(dynamics->XDim());
148  x0_(kP1XIdx) = kP1InitialX;
149  x0_(kP1YIdx) = kP1InitialY;
150  x0_(kP1ThetaIdx) = kP1InitialTheta;
151  x0_(kP1VIdx) = kP1InitialV;
152  x0_(kP2XIdx) = kP2InitialX;
153  x0_(kP2YIdx) = kP2InitialY;
154  x0_(kP2ThetaIdx) = kP2InitialTheta;
155  x0_(kP2VIdx) = kP2InitialV;
156 
157  // Set up initial strategies and operating point. A "Strategy" is a
158  // time-varying affine feedback law for each player, and an "OperatingPoint"
159  // is just a fixed state/control trajectory of the game. Here, we initialize
160  // both to zero (of the correct dimension).
161  strategies_.reset(new std::vector<Strategy>());
162  for (PlayerIndex ii = 0; ii < dynamics->NumPlayers(); ii++)
163  strategies_->emplace_back(kNumTimeSteps, dynamics->XDim(),
164  dynamics->UDim(ii));
165 
166  operating_point_.reset(
167  new OperatingPoint(kNumTimeSteps, dynamics->NumPlayers(), 0.0, dynamics));
168 
169  // Set up costs for all players. These are containers for holding each
170  // player's constituent cost functions and constraints that hold pointwise in
171  // time and can apply to either state or control (for *any* player).
172  // These costs can also build in regularization on the state or the control,
173  // which essentially boils down to adding a scaled identity matrix to each's
174  // Hessian.
175  PlayerCost p1_cost("P1"), p2_cost("P2");
176 
177  // Quadratic control costs.
178  const auto p1_omega_cost = std::make_shared<QuadraticCost>(
179  kOmegaCostWeight, P1::kOmegaIdx, 0.0, "OmegaCost");
180  const auto p1_a_cost = std::make_shared<QuadraticCost>(
181  kACostWeight, P1::kAIdx, 0.0, "AccelerationCost");
182  p1_cost.AddControlCost(0, p1_omega_cost);
183  p1_cost.AddControlCost(0, p1_a_cost);
184 
185  const auto p2_omega_cost = std::make_shared<QuadraticCost>(
186  kOmegaCostWeight, P2::kOmegaIdx, 0.0, "OmegaCost");
187  const auto p2_a_cost = std::make_shared<QuadraticCost>(
188  kACostWeight, P2::kAIdx, 0.0, "AccelerationCost");
189  p2_cost.AddControlCost(1, p2_omega_cost);
190  p2_cost.AddControlCost(1, p2_a_cost);
191 
192  // Constrain each control input to lie in an interval.
193  // Step 3. Try uncommenting these blocks.
194  // const auto p1_omega_max_constraint =
195  // std::make_shared<SingleDimensionConstraint>(
196  // P1::kOmegaIdx, kOmegaMax, false, "Omega Constraint (Max)");
197  // const auto p1_omega_min_constraint =
198  // std::make_shared<SingleDimensionConstraint>(
199  // P1::kOmegaIdx, -kOmegaMax, true, "Omega Constraint (Min)");
200  // const auto p1_a_max_constraint = std::make_shared<SingleDimensionConstraint>(
201  // P1::kAIdx, kAMax, false, "Acceleration Constraint (Max)");
202  // const auto p1_a_min_constraint = std::make_shared<SingleDimensionConstraint>(
203  // P1::kAIdx, -kAMax, true, "Acceleration Constraint (Min)");
204  // p1_cost.AddControlConstraint(0, p1_omega_max_constraint);
205  // p1_cost.AddControlConstraint(0, p1_omega_min_constraint);
206  // p1_cost.AddControlConstraint(0, p1_a_max_constraint);
207  // p1_cost.AddControlConstraint(0, p1_a_min_constraint);
208 
209  // const auto p2_omega_max_constraint =
210  // std::make_shared<SingleDimensionConstraint>(
211  // P2::kOmegaIdx, kOmegaMax, false, "Omega Constraint (Max)");
212  // const auto p2_omega_min_constraint =
213  // std::make_shared<SingleDimensionConstraint>(
214  // P2::kOmegaIdx, -kOmegaMax, true, "Omega Constraint (Min)");
215  // const auto p2_a_max_constraint = std::make_shared<SingleDimensionConstraint>(
216  // P2::kAIdx, kAMax, false, "Acceleration Constraint (Max)");
217  // const auto p2_a_min_constraint = std::make_shared<SingleDimensionConstraint>(
218  // P2::kAIdx, -kAMax, true, "Acceleration Constraint (Min)");
219  // p2_cost.AddControlConstraint(1, p2_omega_max_constraint);
220  // p2_cost.AddControlConstraint(1, p2_omega_min_constraint);
221  // p2_cost.AddControlConstraint(1, p2_a_max_constraint);
222  // p2_cost.AddControlConstraint(1, p2_a_min_constraint);
223 
224  // Encourage each player to go a given nominal speed.
225  const auto p1_nominal_v_cost = std::make_shared<QuadraticCost>(
226  kNominalVCostWeight, kP1VIdx, kP1NominalV, "NominalV");
227  p1_cost.AddStateCost(p1_nominal_v_cost);
228 
229  const auto p2_nominal_v_cost = std::make_shared<QuadraticCost>(
230  kNominalVCostWeight, kP2VIdx, kP2NominalV, "NominalV");
231  p2_cost.AddStateCost(p2_nominal_v_cost);
232 
233  // Encourage each player to remain near the lane center. Could also add
234  // constraints to stay in the lane.
235  const Polyline2 lane1(
236  {Point2(kP1InitialX, -1000.0), Point2(kP1InitialX, 1000.0)});
237  const Polyline2 lane2({Point2(kP2InitialX, 1000.0), Point2(kP2InitialX, 5.0),
238  Point2(kP2InitialX + 5.0, 0.0),
239  Point2(kP2InitialX + 1000.0, 0.0)});
240 
241  const std::shared_ptr<QuadraticPolyline2Cost> p1_lane_cost(
242  new QuadraticPolyline2Cost(kLaneCostWeight, lane1, {kP1XIdx, kP1YIdx},
243  "LaneCenter"));
244  p1_cost.AddStateCost(p1_lane_cost);
245 
246  const std::shared_ptr<QuadraticPolyline2Cost> p2_lane_cost(
247  new QuadraticPolyline2Cost(kLaneCostWeight, lane2, {kP2XIdx, kP2YIdx},
248  "LaneCenter"));
249  p2_cost.AddStateCost(p2_lane_cost);
250 
251  // Penalize proximity (could also use a constraint).
252  constexpr float kMinProximity = 6.0; // m
253  const std::shared_ptr<ProximityCost> p1p2_proximity_cost(
254  new ProximityCost(kProximityCostWeight, {kP1XIdx, kP1YIdx},
255  {kP2XIdx, kP2YIdx}, kMinProximity, "Proximity"));
256  p1_cost.AddStateCost(p1p2_proximity_cost);
257  p2_cost.AddStateCost(p1p2_proximity_cost);
258 
259  // Set up solver.
260  solver_.reset(
261  new ILQSolver(dynamics, {p1_cost, p2_cost}, kTimeHorizon, params));
262 }
