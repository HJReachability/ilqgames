76 std::shared_ptr<SolverLog> ILQSolver::Solve(bool* success, Time max_runtime) {
77  const auto solver_call_time = Clock::now();
78 
79  // Create a new log.
80  std::shared_ptr<SolverLog> log = CreateNewLog();
81 
82  // Last and current operating points. Make sure the last one starts from the
83  // current state so that the current one will start there as well.
84  // NOTE: setting the current operating point to start at x0 is critical to the
85  // constraint satisfaction check at the first iteration.
86  OperatingPoint last_operating_point(problem_->CurrentOperatingPoint());
87  OperatingPoint current_operating_point(problem_->CurrentOperatingPoint());
88  current_operating_point.xs[0] = problem_->InitialState();
89  last_operating_point.xs[0] = problem_->InitialState();
90 
91  // Current strategies.
92  std::vector<Strategy> current_strategies(problem_->CurrentStrategies());
93 
94  // Things to keep track of during each iteration.
95  size_t num_iterations = 0;
96  bool has_converged = false;
97 
98  // Swap operating points and compute new current operating point. Future
99  // operating points will be computed during the call to `ModifyLQStrategies`
100  // which occurs after solving the LQ game.
101  std::vector<float> total_costs;
102  last_operating_point.swap(current_operating_point);
103  CurrentOperatingPoint(last_operating_point, current_strategies,
104  &current_operating_point);
105 
106  // Compute total costs.
107  TotalCosts(current_operating_point, &total_costs);
108 
109  // Log current iterate.
110  Time elapsed = 0.0;
111  log->AddSolverIterate(current_operating_point, current_strategies,
112  total_costs, elapsed, has_converged);
113 
114  // Main loop with timer for anytime execution.
115  while (num_iterations < params_.max_solver_iters && !has_converged &&
116  elapsed < max_runtime - timer_.RuntimeUpperBound()) {
117  // Start loop timer.
118  timer_.Tic();
119 
120  // New iteration.
121  num_iterations++;
122 
123  // Linearize dynamics about the new operating point, only if the system
124  // can't be treated as linear from the outset, in which case we've already
125  // linearized it.
126  // NOTE: we are already computing a new quadraticization
127  // during the linesearch process.
128  if (!problem_->Dynamics()->TreatAsLinear())
129  ComputeLinearization(current_operating_point, &linearization_);
130 
131  // Quadraticize costs..
132  last_cost_quadraticization_.swap(cost_quadraticization_);
133  ComputeCostQuadraticization(current_operating_point,
134  &cost_quadraticization_);
135 
136  // Solve LQ game.
137  current_strategies = lq_solver_->Solve(
138  linearization_, cost_quadraticization_, problem_->InitialState());
139 
140  // Modify this LQ solution.
141  if (!ModifyLQStrategies(&current_strategies, &current_operating_point,
142  &has_converged)) {
143  // Maybe emit warning if exiting early.
144  VLOG(1) << "Solver exited due to linesearch failure.";
145 
146  // Handle success flag.
147  if (success) *success = false;
148 
149  return log;
150  }
151 
152  // Compute total costs and check if we've converged.
153  TotalCosts(current_operating_point, &total_costs);
154 
155  // Record loop runtime.
156  elapsed += timer_.Toc();
157 
158  // Log current iterate.
159  log->AddSolverIterate(current_operating_point, current_strategies,
160  total_costs, elapsed, has_converged);
161  }
162 
163  // Handle success flag.
164  if (success) *success = true;
165 
166  return log;
167 }

169 void ILQSolver::CurrentOperatingPoint(
170  const OperatingPoint& last_operating_point,
171  const std::vector<Strategy>& current_strategies,
172  OperatingPoint* current_operating_point) const {
173  CHECK_NOTNULL(current_operating_point);
174 
175  // Initialize time.
176  current_operating_point->t0 = last_operating_point.t0;
177 
178  // Integrate dynamics and populate operating point, one time step at a time.
179  VectorXf x(last_operating_point.xs[0]);
180  for (size_t kk = 0; kk < time::kNumTimeSteps; kk++) {
181  const Time t = RelativeTimeTracker::RelativeTime(kk);
182 
183  // Unpack.
184  const VectorXf delta_x = x - last_operating_point.xs[kk];
185  const auto& last_us = last_operating_point.us[kk];
186  auto& current_us = current_operating_point->us[kk];
187 
188  // Record state.
189  current_operating_point->xs[kk] = x;
190 
191  // Compute and record control for each player.
192  for (PlayerIndex jj = 0; jj < problem_->Dynamics()->NumPlayers(); jj++) {
193  const auto& strategy = current_strategies[jj];
194  current_us[jj] = strategy(kk, delta_x, last_us[jj]);
195  }
196 
197  // Integrate dynamics for one time step.
198  if (kk < time::kNumTimeSteps - 1)
199  x = problem_->Dynamics()->Integrate(t, time::kTimeStep, x, current_us);
200  }
201 }

215 void ILQSolver::TotalCosts(const OperatingPoint& current_op,
216  std::vector<float>* total_costs) const {
217  // Initialize appropriately.
218  if (total_costs->size() != problem_->PlayerCosts().size())
219  total_costs->resize(problem_->PlayerCosts().size());
220  for (PlayerIndex ii = 0; ii < problem_->PlayerCosts().size(); ii++) {
221  if (problem_->PlayerCosts()[ii].IsTimeAdditive())
222  (*total_costs)[ii] = 0.0;
223  else if (problem_->PlayerCosts()[ii].IsMaxOverTime())
224  (*total_costs)[ii] = -constants::kInfinity;
225  else
226  (*total_costs)[ii] = constants::kInfinity;
227  }
228 
229  // Accumulate costs.
230  for (size_t kk = 0; kk < time::kNumTimeSteps; kk++) {
231  const Time t = RelativeTimeTracker::RelativeTime(kk);
232 
233  for (size_t ii = 0; ii < problem_->PlayerCosts().size(); ii++) {
234  const float current_cost = problem_->PlayerCosts()[ii].Evaluate(
235  t, current_op.xs[kk], current_op.us[kk]);
236 
237  if (problem_->PlayerCosts()[ii].IsTimeAdditive())
238  (*total_costs)[ii] += problem_->PlayerCosts()[ii].Evaluate(
239  t, current_op.xs[kk], current_op.us[kk]);
240  else if (problem_->PlayerCosts()[ii].IsMaxOverTime() &&
241  current_cost > (*total_costs)[ii]) {
242  (*total_costs)[ii] = current_cost;
243  problem_->PlayerCosts()[ii].SetTimeOfExtremeCost(kk);
244  } else if (problem_->PlayerCosts()[ii].IsMinOverTime()) {
245  if (current_cost < (*total_costs)[ii]) {
246  (*total_costs)[ii] = current_cost;
247  problem_->PlayerCosts()[ii].SetTimeOfExtremeCost(kk);
248  }
249  }
250  }
251  }

254 float ILQSolver::StateDistance(const VectorXf& x1, const VectorXf& x2,
255  const std::vector<Dimension>& dims) const {
256  auto total_distance = [&dims](const VectorXf& x1, const VectorXf& x2) {
257  if (dims.empty()) return (x1 - x2).cwiseAbs().maxCoeff();
258 
259  float distance = 0.0;
260  for (const Dimension dim : dims) distance += std::abs(x1(dim) - x2(dim));
261 
262  return distance;
263  }; // total_distance
264 
265  if (problem_->Dynamics()->TreatAsLinear()) {
266  const auto& dyn = problem_->FlatDynamics();
267 
268  // If singular return infinite distance and throw a warning. Otherwise, use
269  // base class implementation but for nonlinear system states.
270  if (dyn.IsLinearSystemStateSingular(x1) ||
271  dyn.IsLinearSystemStateSingular(x2)) {
272  LOG(WARNING)
273  << "Singular state encountered when computing state distance.";
274  return std::numeric_limits<float>::infinity();
275  }
276 
277  return total_distance(dyn.FromLinearSystemState(x1),
278  dyn.FromLinearSystemState(x2));
279  }
280 
281  return total_distance(x1, x2);
282 }

284 bool ILQSolver::ModifyLQStrategies(std::vector<Strategy>* strategies,
285  OperatingPoint* current_operating_point,
286  bool* has_converged) {
287  CHECK_NOTNULL(strategies);
288  CHECK_NOTNULL(current_operating_point);
289  CHECK_NOTNULL(has_converged);
290 
291  // DEBUG: show how alphas are decaying - i.e., we're finding a fixed point.
292  // std::cout << strategies->front().alphas.front().squaredNorm() <<
293  // std::endl;
294 
295  // Precompute expected decrease before we do anything else.
296  SetExpectedDecrease(*strategies);
297 
298  // Initially scale alphas by a fixed amount to avoid unnecessary
299  // backtracking.
300  ScaleAlphas(params_.initial_alpha_scaling, strategies);
301 
302  // Compute next operating point and keep track of whether it satisfies the
303  // Armijo condition.
304  const OperatingPoint last_operating_point(*current_operating_point);
305  float current_stepsize = params_.initial_alpha_scaling;
306  CurrentOperatingPoint(last_operating_point, *strategies,
307  current_operating_point);
308 
309  if (!params_.linesearch) return true;
310 
311  // Keep reducing alphas until we satisfy the Armijo condition.
312  for (size_t ii = 0; ii < params_.max_backtracking_steps; ii++) {
313  // Compute merit function value.
314  const float current_merit_function_value =
315  MeritFunction(*current_operating_point);
316 
317  // Check Armijo condition.
318  if (CheckArmijoCondition(current_merit_function_value, current_stepsize)) {
319  // Success! Update cached terms and check convergence.
320  *has_converged = HasConverged(current_merit_function_value);
321  last_merit_function_value_ = current_merit_function_value;
322  return true;
323  }
324 
325  // Scale down the alphas and try again.
326  ScaleAlphas(params_.geometric_alpha_scaling, strategies);
327  current_stepsize *= params_.geometric_alpha_scaling;
328  CurrentOperatingPoint(last_operating_point, *strategies,
329  current_operating_point);
330  }
331 
332  // Output a warning. Solver should revert to last valid operating point.
333  VLOG(1) << "Exceeded maximum number of backtracking steps.";
334  return false;

337 bool ILQSolver::CheckArmijoCondition(float current_merit_function_value,
338  float current_stepsize) const {
339  // Adjust total expected decrease.
340  const float scaled_expected_decrease =
341  params_.expected_decrease_fraction * current_stepsize *
342  (expected_linear_decrease_ +
343  current_stepsize * expected_quadratic_decrease_);
344 
345  // std::cout << "expected: " << total_expected_decrease << "\n"
346  // << "actual: "
347  // << last_kkt_squared_error_ - *current_kkt_squared_error
348  // << std::endl;
349 
350  return (last_merit_function_value_ - current_merit_function_value >=
351  scaled_expected_decrease);
352 }

354 void ILQSolver::SetExpectedDecrease(
355  const std::vector<Strategy>& current_strategies) {
356  // Compute both linear and quadratic terms from
357  // https://bjack205.github.io/papers/AL_iLQR_Tutorial.pdf (55), but summed for
358  // all players.
359  expected_linear_decrease_ = 0.0;
360  expected_quadratic_decrease_ = 0.0;
361  for (size_t kk = 0; kk < time::kNumTimeSteps; kk++) {
362  for (PlayerIndex ii = 0; ii < problem_->Dynamics()->NumPlayers(); ii++) {
363  const auto& quad = cost_quadraticization_[kk][ii];
364  const auto& alpha = current_strategies[ii].alphas[kk];
365 
366  for (const auto& entry : quad.control) {
367  expected_linear_decrease_ -= entry.second.grad.transpose() * alpha;
368  expected_quadratic_decrease_ +=
369  0.5 * alpha.transpose() * entry.second.hess * alpha;
370  }
371  }
372  }
373 }

375 float ILQSolver::MeritFunction(const OperatingPoint& current_op) {
376  // Accumulate total cost for all players as the merit function.
377  float total_cost = 0.0;
378  for (const auto& pc : problem_->PlayerCosts())
379  total_cost += pc.Evaluate(current_op);
380 
381  return total_cost;

384 void ILQSolver::ComputeLinearization(
385  const OperatingPoint& op,
386  std::vector<LinearDynamicsApproximation>* linearization) {
387  CHECK_NOTNULL(linearization);
388 
389  // Check if linearization is the right length.
390  if (linearization->size() != op.xs.size())
391  linearization->resize(op.xs.size());
392 
393  // Cast dynamics to appropriate type.
394  const auto dyn = static_cast<const MultiPlayerDynamicalSystem*>(
395  problem_->Dynamics().get());
396 
397  // Populate one timestep at a time.
398  for (size_t kk = 0; kk < op.xs.size(); kk++) {
399  const Time t = RelativeTimeTracker::RelativeTime(kk);
400  (*linearization)[kk] = dyn->Linearize(t, op.xs[kk], op.us[kk]);
401  }
402 }

404 void ILQSolver::ComputeLinearization(

406  CHECK_NOTNULL(linearization);

408  // Cast dynamics to appropriate type and make sure the system is
409  // linearizable.
410  CHECK(problem_->Dynamics()->TreatAsLinear());
411  const auto& dyn = problem_->FlatDynamics();
412 
413  // Populate one timestep at a time.
414  for (size_t kk = 0; kk < linearization->size(); kk++)
415  (*linearization)[kk] = dyn.LinearizedSystem();
416 }

418 void ILQSolver::ComputeCostQuadraticization(
419  const OperatingPoint& op,
420  std::vector<std::vector<QuadraticCostApproximation>>* q) {
421  for (size_t kk = 0; kk < time::kNumTimeSteps; kk++) {
422  const Time t = RelativeTimeTracker::RelativeTime(kk);
423  const auto& x = op.xs[kk];
424  const auto& us = op.us[kk];
425 
426  // Quadraticize costs.
427  for (PlayerIndex ii = 0; ii < problem_->Dynamics()->NumPlayers(); ii++) {
428  const PlayerCost& cost = problem_->PlayerCosts()[ii];
429 
430  if (cost.IsTimeAdditive() ||
431  problem_->PlayerCosts()[ii].TimeOfExtremeCost() == kk)
432  (*q)[kk][ii] = cost.Quadraticize(t, x, us);
433  else
434  (*q)[kk][ii] = cost.QuadraticizeControlCosts(t, x, us);
435  }
436  }
