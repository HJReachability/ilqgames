107  GameSolver(const std::shared_ptr<const MultiPlayerIntegrableSystem>& dynamics,
108  const std::vector<PlayerCost>& player_costs, Time time_horizon,
109  const SolverParams& params = SolverParams())
110  : dynamics_(dynamics),
111  player_costs_(player_costs),
112  time_horizon_(time_horizon),
113  time_step_(dynamics->TimeStep()),
114  num_time_steps_(static_cast<size_t>(
115  (constants::kSmallNumber + time_horizon) / time_step_)),
116  linearization_(num_time_steps_),
117  quadraticization_(num_time_steps_),
118  params_(params),
119  timer_(kMaxLoopTimesToRecord) {
120  CHECK_EQ(player_costs_.size(), dynamics_->NumPlayers());
121 
122  if (params_.open_loop)
123  lq_solver_.reset(new LQOpenLoopSolver(dynamics_, num_time_steps_));
124  else
125  lq_solver_.reset(new LQFeedbackSolver(dynamics_, num_time_steps_));
126 
127  // Prepopulate quadraticization.

129  quads.resize(dynamics_->NumPlayers(),
130  QuadraticCostApproximation(dynamics_->XDim()));
131  }
