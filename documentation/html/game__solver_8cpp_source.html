62 // Multiply all alphas in a set of strategies by the given constant.
63 void ScaleAlphas(float scaling, std::vector<Strategy>* strategies) {
64  CHECK_NOTNULL(strategies);
65 
66  for (auto& strategy : *strategies) {
67  for (auto& alpha : strategy.alphas) alpha *= scaling;
68  }
69 }

73 void GameSolver::CurrentOperatingPoint(
74  const OperatingPoint& last_operating_point,
75  const std::vector<Strategy>& current_strategies,
76  OperatingPoint* current_operating_point) const {
77  CHECK_NOTNULL(current_operating_point);
78 
79  // Integrate dynamics and populate operating point, one time step at a time.
80  VectorXf x(last_operating_point.xs[0]);
81  for (size_t kk = 0; kk < num_time_steps_; kk++) {
82  Time t = ComputeTimeStamp(kk);
83 
84  // Unpack.
85  const VectorXf delta_x = x - last_operating_point.xs[kk];
86  const auto& last_us = last_operating_point.us[kk];
87  auto& current_us = current_operating_point->us[kk];
88 
89  // Record state.
90  current_operating_point->xs[kk] = x;
91 
92  // Compute and record control for each player.
93  for (PlayerIndex jj = 0; jj < dynamics_->NumPlayers(); jj++) {
94  const auto& strategy = current_strategies[jj];
95  current_us[jj] = strategy(kk, delta_x, last_us[jj]);
96  }
97 
98  // Integrate dynamics for one time step.
99  if (kk < num_time_steps_ - 1)
100  x = dynamics_->Integrate(t, time_step_, x, current_us);
101  }
102 }

112 bool GameSolver::HasConverged(
113  size_t iteration, const OperatingPoint& last_operating_point,
114  const OperatingPoint& current_operating_point) const {
115  // Check iterations.
116  if (iteration >= params_.max_solver_iters) return true;
117  if (iteration <= 1) return false;
118 
119  // Check operating points.
120  const std::vector<Dimension> kCheckAllDimensions;
121  return AreOperatingPointsClose(last_operating_point, current_operating_point,
122  params_.convergence_tolerance,
123  kCheckAllDimensions);
124 }

126 bool GameSolver::AreOperatingPointsClose(
127  const OperatingPoint& op1, const OperatingPoint& op2, float threshold,
128  const std::vector<Dimension>& dims) const {
129  CHECK_EQ(op1.xs.size(), op2.xs.size());
130 
131  for (size_t kk = 0; kk < op1.xs.size(); kk++) {
132  const VectorXf& x1 = op1.xs[kk];
133  const VectorXf& x2 = op2.xs[kk];
134  if (dims.empty() && (x1 - x2).cwiseAbs().maxCoeff() > threshold)
135  return false;
136  else if (!dims.empty()) {
137  for (const Dimension dim : dims) {
138  if (std::abs(x1(dim) - x2(dim)) > threshold) return false;
139  }
140  }
141  }
142 
143  return true;
144 }

146 bool GameSolver::ModifyLQStrategies(
147  const OperatingPoint& current_operating_point,
148  std::vector<Strategy>* strategies) const {
149  // Compute next operating point.
150  OperatingPoint next_operating_point(num_time_steps_, dynamics_->NumPlayers(),
151  current_operating_point.t0);
152  CurrentOperatingPoint(current_operating_point, *strategies,
153  &next_operating_point);
154 
155  // Initially scale alphas by a fixed amount to avoid unnecessary backtracking.
156  ScaleAlphas(params_.initial_alpha_scaling, strategies);
157 
158  if (!params_.linesearch) return true;
159 
160  // Keep reducing alphas until the maximum elementwise state difference is
161  // above a threshold.
162  for (size_t ii = 0; ii < params_.max_backtracking_steps; ii++) {
163  if (SatisfiesTrustRegion(current_operating_point, next_operating_point))
164  return true;
165 
166  ScaleAlphas(params_.geometric_alpha_scaling, strategies);
167  CurrentOperatingPoint(current_operating_point, *strategies,
168  &next_operating_point);
169  }
170 
171  LOG(WARNING) << "Exceeded maximum number of backtracking steps.";
172  return false;
173 }
