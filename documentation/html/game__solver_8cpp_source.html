66 // Multiply all alphas in a set of strategies by the given constant.
67 void ScaleAlphas(float scaling, std::vector<Strategy>* strategies) {
68  CHECK_NOTNULL(strategies);
69 
70  for (auto& strategy : *strategies) {
71  for (auto& alpha : strategy.alphas) alpha *= scaling;
72  }
73 }

77 bool GameSolver::Solve(const VectorXf& x0,
78  const OperatingPoint& initial_operating_point,
79  const std::vector<Strategy>& initial_strategies,
80  OperatingPoint* final_operating_point,
81  std::vector<Strategy>* final_strategies, SolverLog* log,
82  Time max_runtime) {
83  // Start a stopwatch.
84  const auto solver_call_time = clock::now();
85 
86  // Keep iterating until convergence.
87  auto elapsed_time = [](const std::chrono::time_point<clock>& start) {
88  return std::chrono::duration<Time>(clock::now() - start).count();
89  }; // elapsed_time
90 
91  // Check return pointers not null.
92  CHECK_NOTNULL(final_strategies);
93  CHECK_NOTNULL(final_operating_point);
94 
95  // Make sure we have enough strategies for each time step.
96  DCHECK_EQ(dynamics_->NumPlayers(), initial_strategies.size());
97  DCHECK(std::accumulate(
98  initial_strategies.begin(), initial_strategies.end(), true,
99  [this](bool correct_so_far, const Strategy& strategy) {
100  return correct_so_far &=
101  strategy.Ps.size() == this->num_time_steps_ &&
102  strategy.alphas.size() == this->num_time_steps_;
103  }));
104 
105  // Last and current operating points. Make sure the last one starts from the
106  // current state so that the current one will start there as well.
107  // NOTE: setting the current operating point to start at x0 is critical to the
108  // constraint satisfaction check at the first iteration.
109  OperatingPoint last_operating_point(initial_operating_point);
110  OperatingPoint current_operating_point(initial_operating_point);
111  current_operating_point.xs[0] = x0;
112  last_operating_point.xs[0] = x0;
113 
114  // Current strategies.
115  std::vector<Strategy> current_strategies(initial_strategies);
116 
117  // Reset all constraint barrier weights to unity.
118  for (PlayerCost& cost : player_costs_) cost.ResetConstraintBarrierWeights();
119 
120  // Things to keep track of during each iteration.
121  size_t num_iterations = 0;
122  size_t num_iterations_since_barrier_rescaling = 0;
123  bool has_converged = false;
124 
125  // Turn constraints on/off.
126  auto turn_constraints_on = [this]() {
127  for (auto& cost : player_costs_) cost.TurnConstraintsOn();
128  }; // turn_constraints_on
129 
130  auto turn_constraints_off = [this]() {
131  for (auto& cost : player_costs_) cost.TurnConstraintsOff();
132  }; // turn_constraints_off
133 
134  // Swap operating points and compute new current operating point. Future
135  // operating points will be computed during the call to `ModifyLQStrategies`
136  // which occurs after solving the LQ game.
137  bool was_operating_point_feasible;
138  std::vector<float> total_costs;
139  last_operating_point.swap(current_operating_point);
140  CurrentOperatingPoint(last_operating_point, current_strategies,
141  &current_operating_point, &has_converged, &total_costs,
142  false, &was_operating_point_feasible);
143 
144  // Log current iterate.
145  if (log) {
146  log->AddSolverIterate(current_operating_point, current_strategies,
147  total_costs, elapsed_time(solver_call_time),
148  has_converged);
149  }
150 
151  // Main loop with timer for anytime execution.
152  while (num_iterations < params_.max_solver_iters && !has_converged &&
153  elapsed_time(solver_call_time) <
154  max_runtime - timer_.RuntimeUpperBound()) {
155  // Start loop timer.
156  timer_.Tic();
157 
158  // New iteration.
159  num_iterations++;
160  num_iterations_since_barrier_rescaling++;
161 
162  // Maybe rescale constraint barrier weights.
163  if (num_iterations_since_barrier_rescaling >
164  params_.barrier_scaling_iters) {
165  num_iterations_since_barrier_rescaling = 0;
166  for (PlayerCost& cost : player_costs_)
167  cost.ScaleConstraintBarrierWeights(params_.geometric_barrier_scaling);
168  }
169 
170  // If operating point is feasible, turn on constraints. If it is
171  // not feasible, then turn them off.
172  if (was_operating_point_feasible)
173  turn_constraints_on();
174  else
175  turn_constraints_off();
176 
177  // Linearize dynamics and quadraticize costs for all players about the new
178  // operating point, only if the system can't be treated as linear from the
179  // outset, in which case we've already linearized it.
180  if (!dynamics_->TreatAsLinear())
181  ComputeLinearization(current_operating_point, &linearization_);
182 
183  for (size_t kk = 0; kk < num_time_steps_; kk++) {
184  const Time t = initial_operating_point.t0 + ComputeTimeStamp(kk);
185  const auto& x = current_operating_point.xs[kk];
186  const auto& us = current_operating_point.us[kk];
187 
188  // Quadraticize costs.
189  std::transform(player_costs_.begin(), player_costs_.end(),
190  quadraticization_[kk].begin(),
191  [&t, &x, &us](const PlayerCost& cost) {
192  return cost.Quadraticize(t, x, us);
193  });
194  }
195 
196  // Solve LQ game.
197  current_strategies =
198  lq_solver_->Solve(linearization_, quadraticization_, x0);
199 
200  // Modify this LQ solution.
201  if (!ModifyLQStrategies(&current_strategies, &current_operating_point,
202  &was_operating_point_feasible, &has_converged,
203  &total_costs)) {
204  // Maybe emit warning if exiting early.
205  if (num_iterations == 1) {
206  VLOG(1)
207  << "Solver exited after during first iteration, which may indicate "
208  "an infeasible initial operating point.";
209 
210  if (was_operating_point_feasible)
211  VLOG(1) << "Previous operating point was feasible.";
212  else {
213  VLOG(1) << "Previous operating point was infeasible.";
214  }
215  }
216 
217  return false;
218  }
219 
220  // Log current iterate.
221  if (log) {
222  log->AddSolverIterate(current_operating_point, current_strategies,
223  total_costs, elapsed_time(solver_call_time),
224  has_converged);
225  }
226 
227  // Record loop runtime.
228  timer_.Toc();
229  }
230 
231  CHECK(!player_costs_.front().AreConstraintsOn() ||
232  was_operating_point_feasible);
233 
234  // Maybe emit warning if exiting early.
235  if (num_iterations == 1) {
236  VLOG(1) << "Solver exited after only 1 iteration but passed "
237  "backtracking checks, which may indicate an almost "
238  "converged initial operating point and strategies.";
239  }
240 
241  if (!was_operating_point_feasible) {
242  VLOG(1) << "Solver found an infeasible solution. Failing.";
243  return false;
244  }
245 
246  // Set final strategies and operating point.
247  final_strategies->swap(current_strategies);
248  final_operating_point->swap(current_operating_point);
249 
250  return true;
251 }

253 bool GameSolver::CurrentOperatingPoint(
254  const OperatingPoint& last_operating_point,
255  const std::vector<Strategy>& current_strategies,
256  OperatingPoint* current_operating_point, bool* has_converged,
257  std::vector<float>* total_costs, bool check_trust_region,
258  bool* satisfies_constraints) const {
259  CHECK_NOTNULL(current_operating_point);
260  CHECK_NOTNULL(has_converged);
261  CHECK_NOTNULL(total_costs);
262 
263  current_operating_point->t0 = last_operating_point.t0;
264  *has_converged = true;
265  if (satisfies_constraints) *satisfies_constraints = true;
266  if (total_costs->size() != player_costs_.size())
267  total_costs->resize(player_costs_.size());
268  std::fill(total_costs->begin(), total_costs->end(), 0.0);
269 
270  // Integrate dynamics and populate operating point, one time step at a time.
271  VectorXf x(last_operating_point.xs[0]);
272  for (size_t kk = 0; kk < num_time_steps_; kk++) {
273  const Time t = last_operating_point.t0 + ComputeTimeStamp(kk);
274 
275  // Unpack.
276  const VectorXf delta_x = x - last_operating_point.xs[kk];
277  const auto& last_us = last_operating_point.us[kk];
278  auto& current_us = current_operating_point->us[kk];
279 
280  // Accumulate costs.
281  for (size_t ii = 0; ii < player_costs_.size(); ii++)
282  (*total_costs)[ii] += player_costs_[ii].Evaluate(t, x, current_us);
283 
284  // Check convergence and trust region (including explicit inequality
285  // constraints).
286  auto check_all_constraints = [this](Time t, const VectorXf& x,
287  const std::vector<VectorXf>& us) {
288  for (const auto& cost : this->player_costs_) {
289  if (!cost.CheckConstraints(t, x, us)) return false;
290  }
291  return true;
292  }; // check_all_constraints
293 
294  const float delta_x_distance = StateDistance(
295  x, last_operating_point.xs[kk], params_.trust_region_dimensions);
296  const bool checked_constraints =
297  check_all_constraints(t, x, current_us) || !satisfies_constraints;
298 
299  *has_converged &= delta_x_distance <= params_.convergence_tolerance;
300  if (satisfies_constraints) *satisfies_constraints &= checked_constraints;
301 
302  if (check_trust_region) {
303  if (delta_x_distance > params_.trust_region_size ||
304  (params_.enforce_constraints_in_linesearch && !checked_constraints)) {
305  // If we still satisfy constraints then log a warning. This shouldn't
306  // really ever lead to a fault though since the solver should be
307  // backtracking if this returns false anyway.
308  if (params_.enforce_constraints_in_linesearch && checked_constraints)
309  VLOG(2) << "Failed trust region on time step " << kk
310  << " but satisfied constraints up till then.";
311  return false;
312  }
313  }
314 
315  // Record state.
316  current_operating_point->xs[kk] = x;
317 
318  // Compute and record control for each player.
319  for (PlayerIndex jj = 0; jj < dynamics_->NumPlayers(); jj++) {
320  const auto& strategy = current_strategies[jj];
321  current_us[jj] = strategy(kk, delta_x, last_us[jj]);
322  }
323 
324  // Integrate dynamics for one time step.
325  if (kk < num_time_steps_ - 1)
326  x = dynamics_->Integrate(t, time_step_, x, current_us);
327  }
328 
329  return true;
330 }

332 float GameSolver::StateDistance(const VectorXf& x1, const VectorXf& x2,
333  const std::vector<Dimension>& dims) const {
334  if (dims.empty()) return (x1 - x2).cwiseAbs().maxCoeff();
335 
336  float distance = 0.0;
337  for (const Dimension dim : dims) distance += std::abs(x1(dim) - x2(dim));
338 
339  return distance;
340 }

342 bool GameSolver::ModifyLQStrategies(std::vector<Strategy>* strategies,
343  OperatingPoint* current_operating_point,
344  bool* is_new_operating_point_feasible,
345  bool* has_converged,
346  std::vector<float>* total_costs) const {
347  CHECK_NOTNULL(strategies);
348  CHECK_NOTNULL(current_operating_point);
349  CHECK_NOTNULL(has_converged);
350  CHECK_NOTNULL(total_costs);
351 
352  // Initially scale alphas by a fixed amount to avoid unnecessary
353  // backtracking.
354  ScaleAlphas(params_.initial_alpha_scaling, strategies);
355 
356  // Compute next operating point.
357  const OperatingPoint last_operating_point(*current_operating_point);
358  bool satisfies_trust_region = CurrentOperatingPoint(
359  last_operating_point, *strategies, current_operating_point, has_converged,
360  total_costs, true, is_new_operating_point_feasible);
361 
362  if (!params_.linesearch) return true;
363 
364  // Keep reducing alphas until we satisfy the trust region constraint.
365  for (size_t ii = 0; ii < params_.max_backtracking_steps; ii++) {
366  if (satisfies_trust_region) return true;
367 
368  ScaleAlphas(params_.geometric_alpha_scaling, strategies);
369  satisfies_trust_region = CurrentOperatingPoint(
370  last_operating_point, *strategies, current_operating_point,
371  has_converged, total_costs, true, is_new_operating_point_feasible);
372  }
373 
374  // Output a warning. Solver should revert to last valid operating point.
375  VLOG(1) << "Exceeded maximum number of backtracking steps.";
376  if (!params_.enforce_constraints_in_linesearch) CHECK(!*has_converged);
377  return false;
378 }
