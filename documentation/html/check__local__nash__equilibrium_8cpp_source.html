60 bool NumericalCheckLocalNashEquilibrium(
61  const std::vector<PlayerCost>& player_costs,
62  const std::vector<Strategy>& strategies,
63  const OperatingPoint& operating_point,
64  const MultiPlayerIntegrableSystem& dynamics, const VectorXf& x0,
65  Time time_step, float max_perturbation, bool open_loop) {
66  CHECK_EQ(strategies.size(), player_costs.size());
67  CHECK_EQ(strategies.size(), dynamics.NumPlayers());
68  CHECK_EQ(x0.size(), dynamics.XDim());
69 
70  const size_t num_time_steps = strategies[0].Ps.size();
71  CHECK_EQ(num_time_steps, strategies[0].alphas.size());
72 
73  // Compute nominal equilibrium cost.
74  const std::vector<float> nominal_costs =
75  ComputeStrategyCosts(player_costs, strategies, operating_point, dynamics,
76  x0, time_step, open_loop);
77 
78  // For each player, perturb strategies with Gaussian noise a bunch of times
79  // and if cost decreases then return false.
80  std::vector<Strategy> perturbed_strategies(strategies);
81  for (PlayerIndex ii = 0; ii < dynamics.NumPlayers(); ii++) {
82  for (size_t kk = 0; kk < num_time_steps; kk++) {
83  VectorXf& alphak = perturbed_strategies[ii].alphas[kk];
84 
85  for (size_t jj = 0; jj < alphak.size(); jj++) {
86  alphak(jj) += max_perturbation;
87 
88  // Compute new costs.
89  const std::vector<float> perturbed_costs = ComputeStrategyCosts(
90  player_costs, perturbed_strategies, operating_point, dynamics, x0,
91  time_step, open_loop);
92 
93  // Check Nash condition.
94  if (perturbed_costs[ii] < nominal_costs[ii]) {
95  // std::printf("player %hu, timestep %zu: nominal %f > perturbed
96  // %f\n",
97  // ii, kk, nominal_costs[ii], perturbed_costs[ii]);
98  // std::cout << "nominal u: " <<
99  // operating_point.us[kk][ii].transpose()
100  // << ", alpha original: "
101  // << strategies[ii].alphas[kk].transpose()
102  // << ", vs. perturbed " << alphak.transpose() << std::endl;
103  return false;
104  }
105 
106  // Reset this alpha.
107  alphak = strategies[ii].alphas[kk];
108  }
109 
110  // Reset player ii's strategy.
111  // perturbed_strategies[ii] = strategies[ii];
112  }
113  }
114 
115  return true;
116 }

118 bool CheckSufficientLocalNashEquilibrium(
119  const std::vector<PlayerCost>& player_costs,
120  const OperatingPoint& operating_point, Time time_step,
121  const std::shared_ptr<const MultiPlayerFlatSystem>& dynamics) {
122  // Unpack number of players and number of time steps.
123  const PlayerIndex num_players = player_costs.size();
124  const size_t num_time_steps = operating_point.xs.size();
125  const Dimension xdim = operating_point.xs[0].size();
126 
127  // Set up quadratic cost approximations.
128  std::vector<QuadraticCostApproximation> quadraticization(
129  num_players, QuadraticCostApproximation(xdim));
130 
131  // Quadraticize costs and check PSD conditions.
132  for (size_t kk = 0; kk < num_time_steps; kk++) {
133  const Time t = operating_point.t0 + static_cast<Time>(kk) * time_step;
134  VectorXf x = operating_point.xs[kk];
135  std::vector<VectorXf> us = operating_point.us[kk];
136 
137  if (dynamics.get()) {
138  // Previous x, us are actually xi, vs.
139  x = dynamics->FromLinearSystemState(x.eval());
140  us = dynamics->LinearizingControls(x, std::vector<VectorXf>(us));
141  }
142 
143  std::transform(player_costs.begin(), player_costs.end(),
144  quadraticization.begin(),
145  [&t, &x, &us](const PlayerCost& cost) {
146  return cost.Quadraticize(t, x, us);
147  });
148 
149  // NOTE: we do *not* want to change cost coordinates back to xi, vs because
150  // we're interested in whether the operating point is a local Nash for x, us
151  // (the original problem).
152  // if (dynamics.get()) dynamics->ChangeCostCoordinates(x,
153  // &quadraticization);
154 
155  // Check if Q, Rs PSD.
156  constexpr float kErrorMargin = 1e-4;
157  for (const auto& q : quadraticization) {
158  const auto eig_Q = Eigen::SelfAdjointEigenSolver<MatrixXf>(q.state.hess);
159  if (eig_Q.eigenvalues().minCoeff() < -kErrorMargin) {
160  // std::cout << "Failed at timestep " << kk << std::endl;
161  // std::cout << "Q is: \n" << q.Q << std::endl;
162  // std::cout << "Q evals are: " << eig_Q.eigenvalues().transpose()
163  // << std::endl;
164  return false;
165  }
166 
167  for (const auto& entry : q.control) {
168  const auto eig_R =
169  Eigen::SelfAdjointEigenSolver<MatrixXf>(entry.second.hess);
170  if (eig_R.eigenvalues().minCoeff() < -kErrorMargin) return false;
171  }
172  }
173  }
174 
175  return true;
176 }
