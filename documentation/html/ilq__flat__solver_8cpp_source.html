60 bool ILQFlatSolver::Solve(const VectorXf& xi0,
61  const OperatingPoint& initial_operating_point,
62  const std::vector<Strategy>& initial_strategies,
63  OperatingPoint* final_operating_point,
64  std::vector<Strategy>* final_strategies,
65  SolverLog* log, Time max_runtime) {
66  // Start a stopwatch.
67  const auto solver_call_time = clock::now();
68 
69  // Keep iterating until convergence.
70  auto elapsed_time = [](const std::chrono::time_point<clock>& start) {
71  return std::chrono::duration<Time>(clock::now() - start).count();
72  }; // elapsed_time
73 
74  // Chech return pointers not null.
75  CHECK_NOTNULL(final_strategies);
76  CHECK_NOTNULL(final_operating_point);
77 
78  // Make sure we have enough strategies for each time step.
79  DCHECK_EQ(dynamics_->NumPlayers(), initial_strategies.size());
80  DCHECK(std::accumulate(
81  initial_strategies.begin(), initial_strategies.end(), true,
82  [this](bool correct_so_far, const Strategy& strategy) {
83  return correct_so_far &=
84  strategy.Ps.size() == this->num_time_steps_ &&
85  strategy.alphas.size() == this->num_time_steps_;
86  }));
87 
88  // Cast dynamics.
89  const auto dyn = static_cast<const MultiPlayerFlatSystem*>(dynamics_.get());
90 
91  // Flag for whether or not the initial operating point is all zeros.
92  // If it is all zeros, we will populate it with the initial state unrolled
93  // with the initial strategies in the first iteration of the solver, but
94  // otherwise we will leave it alone.
95  const bool is_initial_operating_point_zero =
96  initial_operating_point.xs[0].squaredNorm() < constants::kSmallNumber;
97 
98  // Last and current operating points.
99  // OperatingPoint last_operating_point(num_time_steps_, dyn->NumPlayers(),
100  // initial_operating_point.t0);
101  OperatingPoint last_operating_point(initial_operating_point);
102  OperatingPoint current_operating_point(initial_operating_point);
103 
104  // Ensure that the current operating point starts at the initial state.
105  current_operating_point.xs[0] = xi0;
106  last_operating_point.xs[0] = xi0;
107 
108  // Current strategies.
109  std::vector<Strategy> current_strategies(initial_strategies);
110 
111  // Preallocate vectors for linearizations and quadraticizations.
112  // Both are time-indexed (and quadraticizations' inner vector is indexed by
113  // player).
114  std::vector<LinearDynamicsApproximation> linearization(
115  num_time_steps_, dyn->LinearizedSystem());
116  std::vector<std::vector<QuadraticCostApproximation>> quadraticization(
117  num_time_steps_);
118  for (auto& quads : quadraticization)
119  quads.resize(dyn->NumPlayers(), QuadraticCostApproximation(dyn->XDim()));
120 
121  // // Preallocate vector of non-linear system controls.
122  // std::vector<VectorXf> us(dyn->NumPlayers());
123  // for (PlayerIndex ii = 0; ii < dyn->NumPlayers(); ii++)
124  // us[ii].resize(dyn->UDim(ii));
125 
126  // Number of iterations, starting from 0.
127  size_t num_iterations = 0;
128 
129  // Log initial iterate.
130  if (log) {
131  log->AddSolverIterate(current_operating_point, current_strategies,
132  EvaluateCosts(current_operating_point),
133  elapsed_time(solver_call_time));
134  }
135 
136  // Keep iterating until convergence.
137  constexpr Time kMaxIterationRuntimeGuess = 2e-2; // s
138  while (elapsed_time(solver_call_time) <
139  max_runtime - kMaxIterationRuntimeGuess &&
140  !HasConverged(num_iterations, last_operating_point,
141  current_operating_point)) {
142  // New iteration.
143  num_iterations++;
144 
145  // auto print_ops = [&last_operating_point,
146  // &current_operating_point](const std::string& msg) {
147  // std::cout << msg << std::endl;
148  // for (size_t kk = 0; kk < last_operating_point.xs.size(); kk++) {
149  // std::cout << kk
150  // << " last x = " << last_operating_point.xs[kk].transpose()
151  // << "\n current x = "
152  // << current_operating_point.xs[kk].transpose() << std::endl;
153  // }
154  // };
155 
156  // Swap operating points and compute new current operating point.
157  if (num_iterations > 1 || is_initial_operating_point_zero) {
158  last_operating_point.swap(current_operating_point);
159  CurrentOperatingPoint(last_operating_point, current_strategies,
160  &current_operating_point);
161  }
162 
163  // Linearize dynamics and quadraticize costs for all players about the new
164  // operating point.
165  for (size_t kk = 0; kk < num_time_steps_; kk++) {
166  const Time t = initial_operating_point.t0 + ComputeTimeStamp(kk);
167  const auto& xi = current_operating_point.xs[kk];
168  const auto& vs = current_operating_point.us[kk];
169 
170  // const VectorXf x = dyn->FromLinearSystemState(xi);
171  // us = dyn->LinearizingControls(x, vs);
172 
173  // Quadraticize costs.
174  std::transform(player_costs_.begin(), player_costs_.end(),
175  quadraticization[kk].begin(),
176  [&t, &xi, &vs](const PlayerCost& cost) {
177  return cost.Quadraticize(t, xi, vs);
178  });
179 
180  // dyn->ChangeControlCostCoordinates(xi, &quadraticization[kk]);
181  }
182 
183  // Solve LQ game.
184  current_strategies =
185  SolveLQGame(*dynamics_, linearization, quadraticization);
186 
187  // Modify this LQ solution.
188  if (!ModifyLQStrategies(current_operating_point, &current_strategies))
189  return false;
190 
191  // Log current iterate.
192  if (log) {
193  log->AddSolverIterate(current_operating_point, current_strategies,
194  EvaluateCosts(current_operating_point),
195  elapsed_time(solver_call_time));
196  }
197  }
198 
199  // Set final strategies and operating point.
200  final_strategies->swap(current_strategies);
201  final_operating_point->swap(current_operating_point);
202 
203  return true;
204 }

206 bool ILQFlatSolver::SatisfiesTrustRegion(
207  const OperatingPoint& last_operating_point,
208  const OperatingPoint& current_operating_point) const {
209  // Check if all states are far from singularity.
210  const auto& dyn = *static_cast<const MultiPlayerFlatSystem*>(dynamics_.get());
211  for (size_t ii = 0; ii < current_operating_point.xs.size(); ii++) {
212  if (dyn.IsLinearSystemStateSingular(current_operating_point.xs[ii])) {
213  return false;
214  }
215  }
216 
217  // Are the operating points close.
218  return AreOperatingPointsClose(last_operating_point, current_operating_point,
219  params_.trust_region_size,
220  params_.trust_region_dimensions);
221 }

223 bool ILQFlatSolver::AreOperatingPointsClose(
224  const OperatingPoint& op1, const OperatingPoint& op2, float threshold,
225  const std::vector<Dimension>& dims) const {
226  CHECK_EQ(op1.xs.size(), op2.xs.size());
227  const auto& dyn = *static_cast<const MultiPlayerFlatSystem*>(dynamics_.get());
228 
229  for (size_t kk = 0; kk < op1.xs.size(); kk++) {
230  VectorXf x1 = op1.xs[kk];
231  VectorXf x2 = op2.xs[kk];
232 
233  // If not singular, use nonlinear system states.
234  if (!dyn.IsLinearSystemStateSingular(x1))
235  x1 = dyn.FromLinearSystemState(x1);
236  if (!dyn.IsLinearSystemStateSingular(x2))
237  x2 = dyn.FromLinearSystemState(x2);
238 
239  if (dims.empty() && (x1 - x2).cwiseAbs().maxCoeff() > threshold)
240  return false;
241  else if (!dims.empty()) {
242  for (const Dimension dim : dims) {
243  if (std::abs(x1(dim) - x2(dim)) > threshold) return false;
244  }
245  }
246  }
247 
248  return true;
249 }
